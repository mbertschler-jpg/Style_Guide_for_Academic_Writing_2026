% ---------------------------------------------------------------------
%  1) FORMATTING SWITCHES  (true / false)
% ---------------------------------------------------------------------

% --- Submission format -----------------------------------------------
%  \digitaltrue  -> DIGITAL submission (Moodle).
%                   Continuous layout, no blank pages, the abridged
%                   title sits in the TOP-RIGHT corner.
%  \digitalfalse -> PRINTED submission (two-sided).
%                   A blank page is inserted after the title page so
%                   that the main text starts on a right-hand page; the
%                   abridged title sits in the TOP-OUTER corner.
\newif\ifdigital
\digitaltrue

% --- Number of authors -----------------------------------------------
%  \pluralfalse -> single author  (I / my / me).
%  \pluraltrue  -> group paper    (We / our / us).
%  Controls the pronouns in the Declaration of Authorship automatically.
\newif\ifplural
\pluralfalse

% --- Optional title-page elements ------------------------------------
\newif\ifshowsubtitle        \showsubtitletrue   % print the subtitle line
\newif\ifshowprogramme       \showprogrammetrue  % print Programme / Module / Course
\newif\ifshowgroupauthors    \showgroupauthorsfalse % print "names of all authors"
\newif\ifshowsecondassessor  \showsecondassessorfalse % print a 2nd assessor
\newif\ifshowsupervisors     \showsupervisorsfalse    % print additional supervisor(s)
\newif\ifshowconfidentiality \showconfidentialityfalse % print confidentiality period
