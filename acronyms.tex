%  sorted alphabetically.
%
%  Syntax:
%     \newacronym{key}{short}{long}
%
%  In-text usage:
%     \gls{key}      -> first use prints  "Long form (SHORT)",
%                       every later use prints just "SHORT".
%     \acrshort{key} -> always the short form (SHORT).
%     \acrlong{key}  -> always the long form.
%     \acrfull{key}  -> always "Long form (SHORT)".
%     \glspl{key}    -> plural ("SHORTs"), first use "Long forms (SHORTs)".
%
%  Do NOT list day-to-day abbreviations that every reader understands,
%  such as "e.g.", "cf." or "etc." -- the guidelines require these to be
%  omitted from the List of Abbreviations.
% =====================================================================
